任务定义¶

方法描述 - 梯度下降¶

$$ : \ Lost_i = \frac 1 2 (y^{\{i\}}_p - y^{\{i\}}_r)^2 $$

  • $y^{\{i\}}_p$ : i(predict), $y^{\{i\}}_p = X_i \times w + b$, $w$
  • $y^{\{i\}}_r$ : i(real)

$$ : \ Cost = \frac 1 m \sum^{m}_{i=1}{Lost_i} $$

  • m :

梯度计算¶

b¶

$\frac {\partial y^{\{i\}}_p}{\partial b} = 1$

$$ \frac {\partial C}{\partial b} = \frac 1 m \sum^{m}_{i=1}{\frac {\partial{L_i}}{\partial b}} = \frac 1 m \sum^{m}_{i=1} [{(y^{\{i\}}_p - y^{\{i\}}_r)} \frac {\partial y^{\{i\}}_p}{\partial b}] = \frac 1 m \sum^{m}_{i=1}{(y^{\{i\}}_p - y^{\{i\}}_r)} $$

w¶

$\frac {\partial y^{\{i\}}_p}{\partial w_j} = X^T_{ij} \rightarrow \frac {\partial y^{\{i\}}_p}{\partial w} = X^T_i$

$$ \frac {\partial C}{\partial w} = \frac 1 m \sum^{m}_{i=1}{\frac {\partial{L_i}}{\partial w}} = \frac 1 m \sum^{m}_{i=1}[{(y^{\{i\}}_p - y^{\{i\}}_r)} \frac {\partial y^{\{i\}}_p}{\partial w}] = \frac 1 m \sum^{m}_{i=1}[{(y^{\{i\}}_p - y^{\{i\}}_r)} X^T_i] $$

# w, b
def calGradient(train_X, Y_predict, Y_real):
    # train_X -- shape = (m, 13)
    # Y_predict -- , shape = (m, 1)
    # Y_real -- , shape = (m, 1)
    assert(Y_predict.shape == Y_real.shape)
    Y_difference = Y_predict - Y_real # y
    db = np.mean(Y_difference) # b
    dw = np.mean(np.dot(train_X.T, Y_difference), axis = 1) # 13
    dw = dw.reshape(-1, 1) # 1
    return dw, db

# Cost
def computeCost(Y_predict, Y_real):
    # Y_predict -- , shape = (m, 1)
    # Y_real -- , shape = (m, 1)
    assert(Y_predict.shape == Y_real.shape)
    losts = np.square(Y_predict - Y_real) / 2 # , 
    cost = np.mean(losts) # , 
    return cost

# w -- shape = (14, 1)
# costs -- list, cost()
def learn(train_X, train_Y, rate, times):
    # train_X -- , shape = (m, 13)
    # train_Y -- , shape = (m, 1)
    # rate    -- 
    # times   -- 
    np.random.seed(1) # 
    assert(type(rate) == float)
    assert(type(times) == int)
    w = np.random.randn(13, 1) * 0.01 / np.sqrt(13) # 
    b = np.random.randn() # 
    costs = []
    for i in range(times):
        Y_predict = np.dot(train_X, w) + b
        Y_real = train_Y
        # Cost
        cost = computeCost(Y_predict, Y_real)
        # print("cost after "+str(i)+" times: "+str(cost))
        costs.append(cost)
        # 
        dw, db = calGradient(train_X, Y_predict, Y_real)
        w -= rate * dw # w
        b -= rate * db # b
    
    b = b.reshape(1,1)
    w = np.row_stack((w, b)) # bw
    assert(w.shape[1] == 1)
    return w, costs

def LRs_predict(X, w):
    # X -- shape = (m, 13)
    # w -- shape = (14, 1)
    #      14? b
    X_row_num = X.shape[0] # X
    # newX.shape = (m, 14);
    # 1

    Y_predict = np.dot(newX, w).reshape(-1, 1) # Y_predict.shape = (m, 1)
    return Y_predict
